```python
# Find the first non-repeating character in a string

def first_unique_char(s):
    char_count = {}
    for char in s:
        char_count[char] = char_count.get(char, 0) + 1
    
    for char in s:
        if char_count[char] == 1:
            return char
    return None

# Test it out!
print(first_unique_char("swiss"))  # Expected output: 'w'
```
